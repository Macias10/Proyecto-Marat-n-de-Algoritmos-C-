#include <iostream>
using namespace std;

int main() {
    int anio;

    // Entrada de datos
    cout << "Ingrese un año: ";
    cin >> anio;

    // Proceso: verificar si es bisiesto
    if ((anio % 4 == 0 && anio % 100 != 0) || (anio % 400 == 0)) {
        cout << "El año " << anio << " es bisiesto." << endl;
    } else {
        cout << "El año " << anio << " no es bisiesto." << endl;
    }

    return 0;
}
