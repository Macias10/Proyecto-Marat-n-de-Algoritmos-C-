#include <iostream>
using namespace std;

int main() {
    int n;
    long long factorial = 1; // Se usa long long para números grandes

    // Entrada de datos
    cout << "Ingrese un número entero positivo: ";
    cin >> n;

    // Validación
    if (n < 0) {
        cout << "Error: no existe el factorial de un número negativo." << endl;
    } else {
        // Proceso: cálculo del factorial
        for (int i = 1; i <= n; i++) {
            factorial *= i;
        }

        // Salida
        cout << "El factorial de " << n << " es: " << factorial << endl;
    }

    return 0;
}
