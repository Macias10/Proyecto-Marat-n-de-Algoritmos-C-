#include <iostream>
using namespace std;

int main() {
    int num1, num2, suma;

    // Pedimos al usuario que ingrese dos números
    cout << "Ingresa el primer número: ";
    cin >> num1;

    cout << "Ingresa el segundo número: ";
    cin >> num2;

    // Calculamos la suma
    suma = num1 + num2;

    // Mostramos el resultado
    cout << "La suma es: " << suma << endl;

    return 0;
}
